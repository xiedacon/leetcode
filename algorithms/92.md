  public ListNode reverseBetween(ListNode head, int m, int n) {
    if (head == null)
      return null;
    ListNode root = new ListNode(0);
    root.next = head;
    ListNode pre = root;
    for (int i = 0; i < m - 1; i++)
      pre = pre.next;

    ListNode start = pre.next;
    ListNode then = start.next;

    for (int i = 0; i < n - m; i++) {
      start.next = then.next;
      then.next = pre.next;
      pre.next = then;
      then = start.next;
    }

    return root.next;
  }
